158 SCENARIO("GeometryInfo construction, copy, and move", "[manifold][!mayfail]")
159 {
160  GIVEN("A SimplicialManifold.")
161  {
162  constexpr auto simplices = static_cast<std::int_fast32_t>(640);
163  constexpr auto timeslices = static_cast<std::int_fast32_t>(4);
164  SimplicialManifold universe(make_triangulation(simplices, timeslices));
165  WHEN("It is constructed.")
166  {
167  THEN("The GeometryInfo struct is not empty.")
168  {
169  CHECK_FALSE(universe.geometry->N3_31() == 0);
170  CHECK_FALSE(universe.geometry->N3_22() == 0);
171  CHECK_FALSE(universe.geometry->N3_13() == 0);
172  CHECK_FALSE(universe.geometry->N1_TL() == 0);
173  CHECK_FALSE(universe.geometry->N1_SL() == 0);
174  CHECK_FALSE(universe.geometry->N0() == 0);
175  }
176  THEN("The GeometryInfo struct matches the Delaunay triangulation.")
177  {
178  CHECK(universe.geometry->number_of_cells() ==
179  universe.triangulation->number_of_finite_cells());
180  CHECK(universe.geometry->number_of_edges() ==
181  universe.triangulation->number_of_finite_edges());
182  CHECK(universe.geometry->N0() ==
183  universe.triangulation->number_of_vertices());
184  }
185  }
186  WHEN("It is copied.")
187  {
188  auto copied_manifold = universe;
189  THEN("The copied manifold's triangulation is valid.")
190  {
191  CHECK(copied_manifold.triangulation->is_valid());
192  CHECK(copied_manifold.triangulation->tds().is_valid());
193  CHECK_FALSE(copied_manifold.geometry == nullptr);
194  }
195  THEN("The copied manifold has the same properties.")
196  {
197  CHECK(copied_manifold.geometry->N3_31() == universe.geometry->N3_31());
198  CHECK(copied_manifold.geometry->N3_22() == universe.geometry->N3_22());
199  CHECK(copied_manifold.geometry->N3_13() == universe.geometry->N3_13());
200  CHECK(copied_manifold.geometry->N1_TL() == universe.geometry->N1_TL());
201  CHECK(copied_manifold.geometry->N1_SL() == universe.geometry->N1_SL());
202  CHECK(copied_manifold.geometry->N0() == universe.geometry->N0());
203  }
204  }
205  WHEN("It is moved.")
206  {
207  auto N3_31_pre_move = universe.geometry->N3_31();
208  auto N3_22_pre_move = universe.geometry->N3_22();
209  auto N3_13_pre_move = universe.geometry->N3_13();
210  auto N1_TL_pre_move = universe.geometry->N1_TL();
211  auto N1_SL_pre_move = universe.geometry->N1_SL();
212  auto N0_pre_move = universe.geometry->N0();
213 
214  auto moved_to_manifold = std::move(universe);
215  THEN("The moved-from manifold is null.")
216  {
217  CHECK(universe.triangulation == nullptr);
218  /// @TODO Why does this persist?
219  CHECK(universe.geometry == nullptr);
220  }
221  THEN("The moved-to manifold is valid.")
222  {
223  CHECK(moved_to_manifold.triangulation->is_valid());
224  CHECK(moved_to_manifold.triangulation->tds().is_valid());
225  CHECK_FALSE(moved_to_manifold.geometry == nullptr);
226  }
227  THEN(
228  "The moved-to manifold has the same properties as the moved-from "
229  "manifold.")
230  {
231  CHECK(moved_to_manifold.geometry->N3_31() == N3_31_pre_move);
232  CHECK(moved_to_manifold.geometry->N3_22() == N3_22_pre_move);
233  CHECK(moved_to_manifold.geometry->N3_13() == N3_13_pre_move);
234  CHECK(moved_to_manifold.geometry->N1_TL() == N1_TL_pre_move);
235  CHECK(moved_to_manifold.geometry->N1_SL() == N1_SL_pre_move);
236  CHECK(moved_to_manifold.geometry->N0() == N0_pre_move);
237  }
238  }
239  }

254  universe.triangulation->number_of_finite_edges());
255  CHECK(universe.geometry->N0() ==

257  CHECK(universe.triangulation->dimension() == 3);
258  CHECK(fix_timeslices(universe.triangulation));
259  CHECK(universe.triangulation->is_valid());
260  CHECK(universe.triangulation->tds().is_valid());
261  // Initial values
262  auto N3_31_pre_swap = universe.geometry->N3_31();
263  auto N3_22_pre_swap = universe.geometry->N3_22();

265  auto N1_TL_pre_swap = universe.geometry->N1_TL();
266  auto N1_SL_pre_swap = universe.geometry->N1_SL();

268 
269  WHEN("It is swapped with a new, empty SimplicialManifold.")

272  REQUIRE(swapped_to_manifold.triangulation->tds().is_valid(true));
273  REQUIRE(swapped_to_manifold.geometry->number_of_cells() == 0);
