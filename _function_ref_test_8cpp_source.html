17 SCENARIO("Lambda operations", "[lambda][!mayfail]")
18 {
19  auto increment_lambda = [](int a) { return ++a; };
20  GIVEN("A simple lambda.")

31  }

33  WHEN("Lambda is called with 5.")
34  {
35  THEN("We should get 6.") { REQUIRE(increment_lambda(5) == 6); }
36  }
37  }
38  GIVEN("A lambda storing a move.")
39  {
40  constexpr auto simplices = static_cast<std::int_fast32_t>(6400);
41  constexpr auto timeslices = static_cast<std::int_fast32_t>(7);
42  SimplicialManifold universe(simplices, timeslices);
43  Move_tracker attempted_moves{{0, 0, 0, 0, 0}};
44  // Verify triangulation
45  CHECK(universe.triangulation);
46  CHECK(universe.geometry->number_of_cells() ==
47  universe.triangulation->number_of_finite_cells());
48  CHECK(universe.geometry->number_of_edges() ==
49  universe.triangulation->number_of_finite_edges());
50  CHECK(universe.geometry->N0() ==
51  universe.triangulation->number_of_vertices());
52  CHECK(universe.triangulation->dimension() == 3);
53  CHECK(fix_timeslices(universe.triangulation));
54  CHECK(universe.triangulation->is_valid());
55  CHECK(universe.triangulation->tds().is_valid());
56 
57  VolumePerTimeslice(universe);
58 
59  CHECK(universe.geometry->max_timevalue().get() == timeslices);
60  CHECK(universe.geometry->min_timevalue().get() == 1);
61  // Previous state
62  auto N3_31_pre_move = universe.geometry->N3_31();
63  auto N3_22_pre_move = universe.geometry->N3_22();
64  auto N3_13_pre_move = universe.geometry->N3_13();
65  auto N1_TL_pre_move = universe.geometry->N1_TL();
66  auto N1_SL_pre_move = universe.geometry->N1_SL();
67  auto N0_pre_move = universe.geometry->N0();
68  // No moves recorded
69  CHECK(attempted_moves[0] == 0);
70 
71  auto move_23_lambda =
72  [](SimplicialManifold manifold,
73  Move_tracker& attempted_moves) -> SimplicialManifold {
74  return make_23_move(std::move(manifold), attempted_moves);
75  };
76  WHEN("The lambda is invoked.")
77  {
78  REQUIRE_NOTHROW(universe = move_23_lambda(universe, attempted_moves));
79  THEN(
80  "The move from the lambda is correct and the triangulation "
81  "invariants are maintained.")
82  {
83  // The triangulation is still valid
84  CHECK(universe.triangulation->tds().is_valid(true));
85  CHECK(universe.triangulation->dimension() == 3);
86  CHECK(fix_timeslices(universe.triangulation));
87  // The move is correct
88  CHECK(universe.geometry->N3_31() == N3_31_pre_move);
89  CHECK(universe.geometry->N3_22() == N3_22_pre_move + 1);
90  CHECK(universe.geometry->N3_13() == N3_13_pre_move);
91  CHECK(universe.geometry->N1_TL() == N1_TL_pre_move + 1);
92  CHECK(universe.geometry->N1_SL() == N1_SL_pre_move);
93  CHECK(universe.geometry->N0() == N0_pre_move);
94  // Move attempts were recorded
95  CHECK_FALSE(attempted_moves[0] == 0);
96  std::cout << "There were " << attempted_moves[0]
97  << " attempted (2,3) moves.\n";
98  }
99  }
100  }
